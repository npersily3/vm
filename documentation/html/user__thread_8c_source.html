170BOOL pageFault(PULONG_PTR arbitrary_va, LPVOID threadContext) {
171
172
173
174 BOOL returnValue;
175 pte* currentPTE;
176 pte pteContents;
177 PCRITICAL_SECTION currentPageTableLock;
178 PTE_REGION* region;
179
180 currentPTE = va_to_pte((ULONG64) arbitrary_va);
181
182
183 // Checks to see if the user is accessing out of bounds va space
184 if (isVaValid((ULONG64) arbitrary_va) == FALSE) {
185 DebugBreak();
186 printf("invalid va %p\n", arbitrary_va);
187 }
188
189 returnValue = !REDO_FAULT;
190 region = getPTERegion(currentPTE);
191 currentPageTableLock = &region->lock;
192
193 EnterCriticalSection(currentPageTableLock);
194 pteContents = *currentPTE;
195
196 // If the page fault was already handled by another thread
197 if (pteContents.validFormat.valid != TRUE) {
198
199 // If the pte is in flight
200 if (isRescue(currentPTE)) {
201 // Determine if the rescue told us to back up or not
202 if (rescue_page((ULONG64) arbitrary_va, currentPTE, (PTHREAD_INFO) threadContext) == REDO_FAULT) {
203 returnValue = REDO_FAULT;
204 }
205 } else {
206 // If we are on a first fault or a disk read map that page.
207 if (mapPage((ULONG64) arbitrary_va, currentPTE, threadContext, currentPageTableLock) == REDO_FAULT) {
208 returnValue = REDO_FAULT;
209 }
210 }
211 }
212
213
214 LeaveCriticalSection(currentPageTableLock);
215
216
217 return returnValue;

237/**
238 * @brief This function will rescue a page table entry that has been unmapped by the trimming thread, but the contents of the virtual page are still on the physical page.
239 * If the page corresponding to the virtual address is on the modified list, this function just simply removes it from the middle.
240 * If the page is on the standby list, this function removes it from the middle of the list and sets the corresponding disk space free.
241 * Lastly, If the page is not on any list, but in the process of being written out, just mark it as no longer being written out.
242 * The writer will see this and free the corresponding disk space
243 *
244 * @param arbitrary_va The virtual address that is faulted on.
245 * @param currentPTE The page table entry corresponding with the virtual address.
246 * @param threadInfo The info of the thread that is faulted on.
247 * @return Returns true if we need to redo the fault, false otherwise.
248 * @pre The page table entry must be locked on entry to this function.
249 * @post The caller must unlock the page table entry
250 */
251BOOL rescue_page(ULONG64 arbitrary_va, pte* currentPTE, PTHREAD_INFO threadInfo) {
252 pfn* page;
253 ULONG64 frameNumber;
254
255 pte entryContents;
256
257 // We must read no fence because the pte could be getting cross edited by another faulter who is using it as a victim
258 entryContents.entireFormat = ReadULong64NoFence((DWORD64 const volatile *)currentPTE);
259
260
261 // The following two checks see if the case where a pte is cross edited in the process of victimization occured
262
263 // If before we got our snapshot, the pte was changed to non rescue
264 if (!isRescue(currentPTE)){
265
266 return REDO_FAULT;
267 }
268
269 frameNumber = entryContents.transitionFormat.frameNumber;
270 page = getPFNfromFrameNumber(frameNumber);
271
272 // if our page that we believe our contents is on no longer maps to our copy
273 if (page->pte->entireFormat != entryContents.entireFormat) {
274 return REDO_FAULT;
275 }
276
277
278
279 enterPageLock(page, threadInfo);
280
281 // Now that we are in the pagelock we can be sure the pte is not being edited without a lock in our victimization code.
282 // Therefore, we can now check the actual pte for the same things as above as opposed to our copy
283
284 if (page->pte != currentPTE) {
285 leavePageLock(page, threadInfo);
286 return REDO_FAULT;
287 }
288
289 if (!isRescue(currentPTE)) {
290
291 leavePageLock(page, threadInfo);
292 return REDO_FAULT;
293 }
294 ASSERT(currentPTE->transitionFormat.frameNumber == frameNumber);
295
296 // If there is a write in progress the page is not on any list
297 // We do not know when in the right this is occurring, so we set this variable to signal to the writer to not add this
298 // page to the standby list
299 // Instead, the writer will free its disk space
300 if (page->isBeingWritten == TRUE) {
301 page->isBeingWritten = FALSE;
302 }
303 // If its on standby, remove it and set the disk space free
304 else if (currentPTE->transitionFormat.contentsLocation == STAND_BY_LIST) {
305
306 removeFromMiddleOfList(&headStandByList, page, threadInfo);
307
308 set_disk_space_free(page->diskIndex);
309
310
311 } else {
312 // It must be on the modified list now if it was determined to be a rescue but no a write in progress or a standby page
313 removeFromMiddleOfList(&headModifiedList,page, threadInfo);
314 }
315 // we can leave this lock because it is no longer on a list, so no other operations like a list removal but be poking at this page
316 // A faulter that is accessing the same region will be stopped by a pagetable lock
317 leavePageLock(page, threadInfo);
318
319 if (MapUserPhysicalPages((PVOID)arbitrary_va, 1, &frameNumber) == FALSE) {
320 DebugBreak();
321 printf("full_virtual_memory_test : could not map VA %p to page %llX\n", (PVOID)arbitrary_va, frameNumber);
322 return FALSE;
323 }
324
325 currentPTE->validFormat.valid = 1;
326 currentPTE->validFormat.transition = UNASSIGNED;
327
328
329 enterPageLock(page, threadInfo);
330 addPageToTail(&headActiveList, page, threadInfo);
331 leavePageLock(page, threadInfo);
332
333 return !REDO_FAULT;
334
335}

631/**
632 * @brief This function gets a page off of one of the free lists. It returns with the page locked.
633 * @param threadContext The thread info of the caller.
634 * @return Returns a free page.
635 * @post The caller must release the page lock of the returned page.
636 */
637pfn* getPageFromFreeList(PTHREAD_INFO threadContext) {
638 ULONG64 localFreeListIndex;
639 PLIST_ENTRY entry;
640 pfn* page;
641 ULONG64 localTotalFreeListLength;
642 boolean pruneInProgress = FALSE;
643 boolean gotFreeListLock = FALSE;
644 ULONG64 counter;
645
646
647
648 counter = 0;
649
650 // Increment global data that approximates what free list should be the most full/least contended
651 localFreeListIndex = InterlockedIncrement(&freeListRemoveIndex) % NUMBER_OF_FREE_LISTS;
652
653
654 // Iterate through the freelists starting at the index calculated above and try to lock it
655
656 while (TRUE) {
657 // In our first run through, just try to enter a freelist lock
658 if (counter < NUMBER_OF_FREE_LISTS) {
659 if (TryEnterCriticalSection(&headFreeLists[localFreeListIndex].page.lock) == TRUE) {
660 gotFreeListLock = TRUE;
661 }
662 // if all of them were empty at the time we checked
663 } else if (counter == 2* NUMBER_OF_FREE_LISTS) {
664 return LIST_IS_EMPTY;
665 // acquire a freelist lock exclusive
666 } else {
667 EnterCriticalSection(&headFreeLists[localFreeListIndex].page.lock);
668 gotFreeListLock = TRUE;
669 }
670
671
672 if (gotFreeListLock) {
673#if DBG
674 validateList(&headFreeLists[localFreeListIndex]);
675#endif
676 // now we have a locked page
677 entry = RemoveHeadList(&headFreeLists[localFreeListIndex]);
678
679 if (entry == LIST_IS_EMPTY) {
680 page = LIST_IS_EMPTY;
681 } else {
682 page = container_of(entry, pfn, entry);
683 }
684#if DBG
685 validateList(&headFreeLists[localFreeListIndex]);
686#endif
687
688 // now that the page is off the list we no longer need to lock the list
689 LeaveCriticalSection(&headFreeLists[localFreeListIndex].page.lock);
690
691 if (page == LIST_IS_EMPTY) {
692 gotFreeListLock = FALSE;
693 counter++;
694 localFreeListIndex = (localFreeListIndex + 1) % NUMBER_OF_FREE_LISTS;
695 continue;
696 } else {
697
698 // The following code calls the standby pruner.
699 localTotalFreeListLength = InterlockedDecrement64((volatile LONG64 *) &freeListLength);
700
701 ASSERT(localTotalFreeListLength != MAXULONG64);
702 // check to see if we have enough pages on the free list
703 if (localTotalFreeListLength <= STAND_BY_TRIM_THRESHOLD) {
704 // if there is not a pruning mission already happening
705 pruneInProgress = InterlockedCompareExchange((volatile LONG *)&standByPruningInProgress, TRUE, FALSE);
706
707 if (pruneInProgress == FALSE) {
708 batchVictimsFromStandByList(threadContext);
709 }
710
711 InterlockedExchange(&standByPruningInProgress, FALSE);
712 }
713 // return a locked page
714 enterPageLock(page, threadContext);
715 return page;
716 }
717 }
718
719 // go around in a circle
720 localFreeListIndex = (localFreeListIndex + 1) % NUMBER_OF_FREE_LISTS;
721 counter++;
722 }
