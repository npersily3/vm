129 (currentPTE->transitionFormat.contentsLocation != STAND_BY_LIST )) {
130
131 leavePageLock(page, threadInfo);
132 return REDO_FREE;
133 }
134 ASSERT(currentPTE->transitionFormat.frameNumber == frameNumber);
135 //if there is a write in progress
136 if (page->isBeingWritten == TRUE) {
137 page->isBeingWritten = FALSE;
138 page->isBeingFreed = TRUE;
139 leavePageLock(page, threadInfo);
140 *addToFreeList = FALSE;
141 }
142 else if (currentPTE->transitionFormat.contentsLocation == STAND_BY_LIST) {
143
144 removeFromMiddleOfList(&headStandByList, page, threadInfo);
145 set_disk_space_free(page->diskIndex);
146 *addToFreeList = TRUE;
147
148 } else {
149 removeFromMiddleOfList(&headModifiedList,page, threadInfo);
150 *addToFreeList = TRUE;
151 }
152
153
154
155 return !REDO_FREE;
