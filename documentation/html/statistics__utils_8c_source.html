117double getSlope(double* intervals, ULONG64 count) {
118
119
120 double sum_x = 0;
121 double sum_y = 0;
122 double sum_xy = 0;
123 double sum_x2 = 0;
124
125 double x;
126 double y;
127 for (int i = 0; i < count; i++) {
128 x = (double)i; // Time index
129 y = (double) intervals[i]; // Interval length
130
131 sum_x += x;
132 sum_y += y;
133 sum_xy += x * y;
134 sum_x2 += x * x;
135 }
136
137 double n = (float)count;
138 double slope = (n * sum_xy - sum_x * sum_y) / (n * sum_x2 - sum_x * sum_x);
139
140 return slope;
141
142}
