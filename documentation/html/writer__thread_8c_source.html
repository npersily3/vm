19/**
20 * @brief This function takes pages off the modified list and tries to write them to disk. It batch writes pages to disk to be more effiecient.
21 * @param info A pointer to a thread info struct. Passed in during the function CreateThread
22 * @retval 0 If the program succeeds
23 */
24DWORD diskWriter (LPVOID info) {
25
26
27 ULONG64 localBatchSize;
28 ULONG64 previousBatchSize;
29
30 ULONG64 diskAddressArray[BATCH_SIZE];
31 ULONG64 diskIndexArray[BATCH_SIZE];
32
33 ULONG64 frameNumberArray[BATCH_SIZE];
34 pfn* pfnArray[BATCH_SIZE];
35
36 PTHREAD_INFO threadContext = (PTHREAD_INFO)info;
37
38 HANDLE* events[2];
39 DWORD returnEvent;
40 events[0] = writingStartEvent;
41 events[1] = systemShutdownEvent;
42
43 while (TRUE) {
44
45
46
47 returnEvent = WaitForMultipleObjects(2,events, FALSE, INFINITE);
48
49 //if the system shutdown event was signaled, exit
50 if (returnEvent - WAIT_OBJECT_0 == 1) {
51 return 0;
52 }
53
54 localBatchSize = BATCH_SIZE;
55
56 // pre acquire all the disk slots
57 localBatchSize = getMultipleDiskIndices(diskIndexArray);
58
59 // the disk was empty upon entry
60 if (localBatchSize == COULD_NOT_FIND_SLOT) {
61 SetEvent(writingEndEvent);
62 continue;
63 }
64
65 // save how many disk indices we got
66 previousBatchSize = localBatchSize;
67
68 // fill our local page and frame number arrays
69 getPagesFromModifiedList(&localBatchSize, pfnArray, diskIndexArray, frameNumberArray, threadContext);
70
71
72
73 // if the amount of pages we got off the modified list is less than the amount of disk slots we acquired free the disk slots
74 if (previousBatchSize != localBatchSize) {
75 freeUnusedDiskSlots(diskIndexArray, localBatchSize, previousBatchSize);
76 }
77
78 if (localBatchSize == 0) {
79 SetEvent(writingEndEvent);
80 continue;
81 }
82
83 // simply converts indices to addresses
84 getDiskAddressesFromDiskIndices(diskIndexArray, diskAddressArray, localBatchSize);
85
86 // actual write to disk
87 writeToDisk(localBatchSize, frameNumberArray, diskAddressArray);
88
89 addToStandBy(localBatchSize, pfnArray, threadContext);
90
91// Usually it is fine to have stale data stored in these stack variables,
92// but for debugging it is useful to reset the variables
93#if DBG
94for (int i = 0; i < BATCH_SIZE; ++i)
95{
96 diskAddressArray[i] = 0;
97 diskIndexArray[i] = 0;
98 frameNumberArray[i] = 0;
99 pfnArray[i] = 0;
100
101}
102
103#endif
104
105
106 // this needs to be in a sharedLock to avoid
107 // the race condition where it is reset right after it is
108 // TODO think about this
109 acquire_srw_exclusive(&headStandByList.sharedLock, (PTHREAD_INFO) threadContext);
110 SetEvent(writingEndEvent);
111 release_srw_exclusive(&headStandByList.sharedLock);
112 }
113
114}

128/**
129 *
130 *@brief Adds pages from an array onto the standby list.
131 * @param localBatchSize The number of pages to add
132 * @param pfnArray An array of page pointers
133 * @param info The info about the caller
134 */
135VOID addToStandBy(ULONG64 localBatchSize, pfn** pfnArray, PTHREAD_INFO info) {
136
137 pfn* page;
138
139
140 for (int i = 0; i < localBatchSize; ++i) {
141
142 page = pfnArray[i];
143
144
145 enterPageLock(page, info);
146
147 //if it has been rescued, free up the disk space and do not put it on the standby list because the faulter
148 // will put it on the active list
149 if(page->isBeingWritten == FALSE) {
150 set_disk_space_free(page->diskIndex);
151
152 // if a pte/page pair had been freed, the user thread does not want to wait for the writer thread
153 // to finish in order to release locks, so the faulter signals this status and tells us to put this page on
154 // the freelist
155 if (page->isBeingFreed == TRUE) {
156 page->isBeingFreed = FALSE;
157 addPageToFreeList(page, info);
158 }
159 } else {
160
161 page->isBeingWritten = FALSE;
162
163 // this expects me to come in with the lock and does not release it for me
164 addPageToTail(&headStandByList, page, info);
165
166 }
167
168 leavePageLock(page, info);
169 }
170}

200/**
201 * @brief This function pops pages off the modified list and updates their status as standby, in progress write, and gives them a disk index
202 * @param localBatchSizePointer A pointer to our local batch size. If there are less pages on the modified list than disk indices,
203 * the function will update the contents of the pointer to reflect that change.
204 * @param pfnArray A location for this function to store the pages it pops of the modified list.
205 * @param diskIndexArray An array of disk indices to update the pfns.
206 * @param frameNumberArray An array of frame numbers to be filled in.
207 * @param threadContext The thread info of the caller.
208 */
209VOID getPagesFromModifiedList (PULONG64 localBatchSizePointer, pfn** pfnArray, PULONG64 diskIndexArray, PULONG64 frameNumberArray, PTHREAD_INFO threadContext) {
210
211 ULONG64 i;
212 pfn* page;
213 ULONG64 frameNumber;
214 BOOL doubleBreak;
215
216
217 i = 0;
218 doubleBreak = FALSE;
219
220 for (; i < *localBatchSizePointer; i++) {
221
222
223 // this function obtains the page lock for us
224 page = RemoveFromHeadofPageList(&headModifiedList, threadContext);
225
226
227 // if we reach the end update the contents of the size pointer and break
228 if (page == LIST_IS_EMPTY) {
229 if (i == 0) {
230 doubleBreak = TRUE;
231 }
232 *localBatchSizePointer = i;
233 break;
234 }
235 frameNumber = getFrameNumber(page);
236
237 pfnArray[i] = page;
238 frameNumberArray[i] = frameNumber;
239
240 updatePage(page, diskIndexArray[i]);
241
242 leavePageLock(page, threadContext);
243 }
244 if (doubleBreak == TRUE) {
245 return;
246 }
247
248 return;
249}
