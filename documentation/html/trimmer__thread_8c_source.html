21/**
22 * @brief This is the function that deals with unmapping pages.
23 * It will pop a page off the active list and add it to the modified list.
24 * It will batch unmap if the consecutive pages are in the same page table entry region.
25 *
26 * @param info A pointer to a thread info struct. Passed in during the function CreateThread
27 * @retval 0 If the program succeeds
28 */
29DWORD page_trimmer(LPVOID info) {
30 ULONG64 BatchIndex;
31 BOOL doubleBreak;
32 PCRITICAL_SECTION trimmedPageTableLock;
33 pfn* page;
34 pfn* pages[BATCH_SIZE];
35 ULONG64 virtualAddresses[BATCH_SIZE];
36 PTE_REGION* region;
37 PTHREAD_INFO threadContext;
38 threadContext = (PTHREAD_INFO)info;
39
40 HANDLE events[2];
41 DWORD returnEvent;
42 events[0] = trimmingStartEvent;
43 events[1] = systemShutdownEvent;
44
45
46 while (TRUE) {
47 BatchIndex = 0;
48 doubleBreak = FALSE;
49 trimmedPageTableLock = NULL;
50
51 returnEvent = WaitForMultipleObjects(2, events, FALSE, INFINITE);
52
53 //if the system shutdown event was signaled, exit
54 if (returnEvent - WAIT_OBJECT_0 == 1) {
55 return 0;
56 }
57
58 // while there is still stuff to trim and the arrays are not at capacity
59 while (headModifiedList.length < NUMBER_OF_PHYSICAL_PAGES / 4 && BatchIndex < BATCH_SIZE) {
60
61 page = getActivePage(threadContext);
62
63 if (page == NULL) {
64 if (BatchIndex == 0) {
65 doubleBreak = TRUE;
66 }
67 break;
68 }
69
70 // if we are not on a streak we need to get a new region a lock the ptes
71 if (trimmedPageTableLock == NULL) {
72
73 region = getPTERegion(page->pte);
74 trimmedPageTableLock = &region->lock;
75 EnterCriticalSection(trimmedPageTableLock);
76 } else {
77 ASSERT(region == getPTERegion(page->pte))
78 }
79
80
81 virtualAddresses[BatchIndex] = (ULONG64) pte_to_va(page->pte);
82 pages[BatchIndex] = page;
83 pages[BatchIndex]->pte->transitionFormat.mustBeZero = 0;
84 pages[BatchIndex]->pte->transitionFormat.contentsLocation = MODIFIED_LIST;
85 BatchIndex++;
86
87 if (isNextPageInSameRegion(region, threadContext) == FALSE) {
88 unmapBatch(virtualAddresses, BatchIndex);
89 addBatchToModifiedList(pages, BatchIndex, threadContext);
90
91 LeaveCriticalSection(trimmedPageTableLock);
92 // Setting this to null tells the next loop to get a new region
93 trimmedPageTableLock = NULL;
94 BatchIndex = 0;
95
96 }
97 }
98 if (doubleBreak == TRUE) {
99 SetEvent(writingStartEvent);
100 continue;
101 }
102 if (BatchIndex == 0) {
103 SetEvent(writingStartEvent);
104 continue;
105 }
106
107 //this just handles the last batch
108 unmapBatch(virtualAddresses, BatchIndex);
109 addBatchToModifiedList(pages, BatchIndex, (PTHREAD_INFO) threadContext);
110
111 LeaveCriticalSection(trimmedPageTableLock);
112
113 //nptodo add a condition around this
114 SetEvent(writingStartEvent);
115
116
117 }
118}

161VOID addBatchToModifiedList (pfn** pages, ULONG64 batchSize, PTHREAD_INFO threadContext) {
162
163 pfn* page;
164
165 for (int i = 0; i < batchSize; ++i) {
166 page = pages[i];
167 enterPageLock(page, threadContext);
168 addPageToTail(&headModifiedList, page, threadContext);

173/**
174 * @brief This function peeks into the head of the active list and sees if the next page is in the same pte region.
175 * @param region The page table region struct of the batch that is currently being trimmed.
176 * @param info The info of the caller.
177 * @return Returns true if the next page's corresponding page table entry is in the region passed in.
178 */
179BOOL isNextPageInSameRegion(PTE_REGION* region, PTHREAD_INFO info) {
180
181 pfn* nextPage;
182 PTE_REGION* nextRegion;
183 sharedLock* lock;
184 lock = &headActiveList.sharedLock;
185
186 // Peek ahead
187 // TODO make this a no fence on the entry
188 enterPageLock(&headActiveList.page, info);
189 nextPage = container_of(headActiveList.entry.Flink, pfn, entry);
190 leavePageLock(&headActiveList.page, info);
191
192
193 if (&headActiveList.entry == &nextPage->entry) {
194 return LIST_IS_EMPTY;
195 }
196
197 nextRegion = getPTERegion(nextPage->pte);
198
199 return nextRegion == region;
